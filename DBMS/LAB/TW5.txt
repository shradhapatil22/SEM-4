CREATE TABLE STATEMENTS

CREATE TABLE ACTOR(
	ACT_ID INTEGER,
	ACT_NAME VARCHAR(15),
	ACT_GENDER VARCHAR(15),
	PRIMARY KEY(ACT_ID)
);

CREATE TABLE DIRECTOR(
	DIR_ID INTEGER,
	DIR_NAME VARCHAR(15),
	DIR_PHONE INTEGER,
	PRIMARY KEY(DIR_ID)
);

CREATE TABLE MOVIES(
	MOV_ID INTEGER,
	MOV_TITLE VARCHAR(15),
	MOV_YEAR INTEGER,
	MOV_LANG VARCHAR(15),
	DIR_ID INTEGER,
	PRIMARY KEY(MOV_ID),
	FOREIGN KEY(DIR_ID) REFERENCES DIRECTOR(DIR_ID) 
);

CREATE TABLE MOVIE_CASTE(
	ACT_ID INTEGER,
	MOV_ID INTEGER,
	FOREIGN KEY(ACT_ID) REFERENCES ACTOR(ACT_ID) ,
	FOREIGN KEY(MOV_ID) REFERENCES MOVIES(MOV_ID) ,
	ROLE VARCHAR(15)
);


CREATE TABLE RATING(
	MOV_ID INTEGER,
	FOREIGN KEY(MOV_ID) REFERENCES MOVIES(MOV_ID),
	REV_STARS INTEGER
);

SELECTION STATEMENTS:
